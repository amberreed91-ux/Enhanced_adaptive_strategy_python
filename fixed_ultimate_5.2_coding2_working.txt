f_calc_atr(int len) =>
    ta.rma(ta.tr(true), len)

f_calc_efficiency(float src, int len) =>
    float change_val = math.abs(src - src[len])
    float volatility_sum = math.sum(math.abs(src - src[1]), len)
    float result = volatility_sum > 0 ? change_val / volatility_sum : 0.0
    result

f_is_inside_zone(float price, float zone_top, float zone_bottom) =>
    price <= zone_top and price >= zone_bottom

f_mtf_maz_bias(float htf_close, float htf_sma, float htf_atr, float deadband) =>
    float diff = htf_close - htf_sma
    float threshold = htf_atr * deadband
    int result = diff > threshold ? 1 : diff < -threshold ? -1 : 0
    result

f_is_time_in_range(int _hour, int _min, int start_h, int start_m, int end_h, int end_m) =>
    int current_minutes = (_hour * 60) + _min
    int start_minutes   = (start_h * 60) + start_m
    int end_minutes     = (end_h * 60) + end_m
    bool result = false
    if end_minutes < start_minutes
        result := (current_minutes >= start_minutes) or (current_minutes <= end_minutes)
    else
        result := (current_minutes >= start_minutes) and (current_minutes <= end_minutes)
    result

f_bb_width_percent(float src, int len, float mult) =>
    [_basis, upper, lower] = ta.bb(src, len, mult)
    float width = upper - lower
    float sma = ta.sma(src, len)
    float percent = width / sma * 100
    percent

f_calc_performance_score(int lookback) =>
    float result = 0.0
    if lookback > 0 and lookback <= 500 and bar_index >= lookback
        float current_equity = strategy.equity
        float past_equity    = strategy.equity[lookback]
        if not na(past_equity) and past_equity > 0 and not na(current_equity) and current_equity > 0
            float total_return  = ((current_equity - past_equity) / past_equity) * 100
            float equity_stdev  = ta.stdev(strategy.equity, lookback)
            float volatility_eq = (equity_stdev / past_equity) * 100
            float max_dd = 0.0
            float peak   = past_equity
            for i = 0 to lookback - 1
                float eq_i = strategy.equity[i]
                if not na(eq_i)
                    if eq_i > peak
                        peak := eq_i
                    float dd = (peak - eq_i) / peak * 100
                    if dd > max_dd
                        max_dd := dd
            float denominator = volatility_eq + max_dd + 1.0
            result := denominator > 0 ? total_return / denominator : total_return
    result

f_optimize_atr_mult(float _atr, float _atr_mult_max, float _atr_mult_min, bool _defensive) =>
    float vol_perc    = ta.percentrank(_atr, 100)
    float optimal_mult = _atr_mult_max - ((_atr_mult_max - _atr_mult_min) * vol_perc / 100)
    _defensive ? optimal_mult * 0.7 : optimal_mult

f_optimize_confluence_threshold(int _conf_max, int _conf_min, bool _defensive) =>
    int recent_trades = math.min(50, strategy.closedtrades)
    float win_rate = 50.0
    if recent_trades > 10
        int wins = 0
        for i = 0 to recent_trades - 1
            if strategy.closedtrades.profit(i) > 0
                wins := wins + 1
        win_rate := (wins * 100.0) / recent_trades
    float threshold_adj = (50.0 - win_rate) * 0.5
    int optimal_threshold = int(math.max(_conf_min, math.min(_conf_max, 70 + threshold_adj)))
    _defensive ? math.min(85, optimal_threshold + 10) : optimal_threshold

f_optimize_position_risk(float _risk_max, float _risk_min, bool _defensive) =>
    int consec_losses_local = 0
    for i = 0 to math.min(10, strategy.closedtrades) - 1
        if strategy.closedtrades.profit(i) > 0
            break
        consec_losses_local := consec_losses_local + 1
    float risk_mult    = math.max(0.25, 1.0 - (consec_losses_local * 0.15))
    float optimal_risk = _risk_max - ((_risk_max - _risk_min) * (1.0 - risk_mult))
    _defensive ? optimal_risk * 0.5 : optimal_risk

f_optimize_rb_boost(int _htf_bias_count, int _regime_persist, string _cur_regime, bool _defensive) =>
    int stability_score = _htf_bias_count >= _regime_persist ? 2 : (_htf_bias_count >= 2 ? 1 : 0)
    int boost = 0
    if _cur_regime == "TRENDING"
        boost := stability_score
    else if _cur_regime == "RANGING"
        boost := stability_score + 1
    _defensive ? 0 : boost

f_update_portfolio_correlations() =>
    float _c1   = portfolio_correlation_1
    float _c2   = portfolio_correlation_2
    float _c3   = portfolio_correlation_3
    float _avg  = portfolio_avg_correlation
    float _w1   = hrp_weight_1
    float _w2   = hrp_weight_2
    float _w3   = hrp_weight_3
    float _heat = portfolio_heat_score
    float _radj = portfolio_risk_adjustment
    bool  _dok  = portfolio_diversification_ok
    string _rst = portfolio_risk_status
    int   _cnt  = correlated_assets_count
    if portfolio_mode
        float close_1 = request.security(correlation_symbol_1, timeframe.period, close, lookahead=barmerge.lookahead_off)
        float close_2 = request.security(correlation_symbol_2, timeframe.period, close, lookahead=barmerge.lookahead_off)
        float close_3 = request.security(correlation_symbol_3, timeframe.period, close, lookahead=barmerge.lookahead_off)
        float ret_1 = (close_1 - close_1[1]) / close_1[1]
        float ret_2 = (close_2 - close_2[1]) / close_2[1]
        float ret_3 = (close_3 - close_3[1]) / close_3[1]
        _c1 := nz(ta.correlation(ret_1, ret_2, corr_lookback), 0)
        _c2 := nz(ta.correlation(ret_1, ret_3, corr_lookback), 0)
        _c3 := nz(ta.correlation(ret_2, ret_3, corr_lookback), 0)
        _avg := (math.abs(_c1) + math.abs(_c2) + math.abs(_c3)) / 3.0
        float var_1 = ta.variance(ret_1, corr_lookback)
        float var_2 = ta.variance(ret_2, corr_lookback)
        float var_3 = ta.variance(ret_3, corr_lookback)
        float inv_1 = var_1 > 0 ? 1.0 / var_1 : 1.0
        float inv_2 = var_2 > 0 ? 1.0 / var_2 : 1.0
        float inv_3 = var_3 > 0 ? 1.0 / var_3 : 1.0
        float tot_inv = inv_1 + inv_2 + inv_3
        if tot_inv > 0
            _w1 := inv_1 / tot_inv
            _w2 := inv_2 / tot_inv
            _w3 := inv_3 / tot_inv
        _heat := _avg * 100
        _cnt  := (math.abs(_c1) > max_correlation_threshold ? 1 : 0) + (math.abs(_c2) > max_correlation_threshold ? 1 : 0) + (math.abs(_c3) > max_correlation_threshold ? 1 : 0)
        _radj := _heat < 40 ? 1.0 : _heat < 60 ? 0.85 : _heat < 80 ? 0.7 : 0.5
        _dok  := _heat < 60
        _rst  := _heat < 40 ? "GREEN" : _heat < 60 ? "YELLOW" : "RED"
    [_c1, _c2, _c3, _avg, _w1, _w2, _w3, _heat, _radj, _dok, _rst, _cnt]

f_online_learning_update() =>
    float _lerr  = ol_learning_error
    float _cerr  = ol_cumulative_error
    int   _ucnt  = ol_update_count
    float _arate = ol_adaptation_rate
    float _drift = ol_concept_drift_score
    bool  _adapt = ol_is_adapting
    float _wt    = ol_regime_weights_trend
    float _wr    = ol_regime_weights_range
    float _wv    = ol_regime_weights_vol
    float _fw1   = ol_feature_weights_1
    float _fw2   = ol_feature_weights_2
    float _fw3   = ol_feature_weights_3
    float _fw4   = ol_feature_weights_4
    float _mb1   = ol_memory_buffer_1
    float _mb2   = ol_memory_buffer_2
    float _mb3   = ol_memory_buffer_3
    float _tprob = ol_regime_transition_prob
    if online_learning_enabled and bar_index % ol_update_frequency == 0
        float actual_return = (close - close[1]) / close[1]
        float predicted = current_regime == "TRENDING" ? 0.001 : current_regime == "RANGING" ? 0.0 : -0.001
        _lerr  := actual_return - predicted
        _cerr  := _cerr * ol_forgetting_factor + math.abs(_lerr)
        _ucnt  := _ucnt + 1
        _arate := _cerr / _ucnt
        float error_change = math.abs(_lerr) - _arate
        _drift := error_change > ol_adaptation_threshold ? 1.0 : 0.0
        _adapt := _drift > 0
        if _adapt
            if current_regime == "TRENDING"
                _wt := math.min(0.6, _wt + ol_learning_rate)
            else if current_regime == "RANGING"
                _wr := math.min(0.6, _wr + ol_learning_rate)
            else
                _wv := math.min(0.6, _wv + ol_learning_rate)
            float total_weight = _wt + _wr + _wv
            if total_weight > 0
                _wt := _wt / total_weight
                _wr := _wr / total_weight
                _wv := _wv / total_weight
        float rsi_val = ta.rsi(close, 14)
        float atr_val = f_calc_atr(14)
        float mom_val = close - close[10]
        float feature_1 = rsi_val / 100.0
        float feature_2 = math.min(1.0, atr_val / close * 100)
        float feature_3 = volume / ta.sma(volume, 20)
        float feature_4 = mom_val / close
        float prediction = _fw1 * feature_1 + _fw2 * feature_2 + _fw3 * feature_3 + _fw4 * feature_4
        float error = actual_return - prediction
        _fw1 := _fw1 + ol_learning_rate * error * feature_1 * ol_forgetting_factor
        _fw2 := _fw2 + ol_learning_rate * error * feature_2 * ol_forgetting_factor
        _fw3 := _fw3 + ol_learning_rate * error * feature_3 * ol_forgetting_factor
        _fw4 := _fw4 + ol_learning_rate * error * feature_4 * ol_forgetting_factor
        float feat_total = _fw1 + _fw2 + _fw3 + _fw4
        if feat_total > 0
            _fw1 := _fw1 / feat_total
            _fw2 := _fw2 / feat_total
            _fw3 := _fw3 / feat_total
            _fw4 := _fw4 / feat_total
        _mb3   := _mb2
        _mb2   := _mb1
        _mb1   := actual_return
        _tprob := math.abs(_mb1 - _mb2) + math.abs(_mb2 - _mb3)
    [_lerr, _cerr, _ucnt, _arate, _drift, _adapt, _wt, _wr, _wv, _fw1, _fw2, _fw3, _fw4, _mb1, _mb2, _mb3, _tprob]

f_ml_ensemble_vote() =>
    float rsi_val = ta.rsi(close, 14)
    float atr_ratio = f_calc_atr(14) / ta.sma(f_calc_atr(14), 20)
    float momentum = (close - close[10]) / close[10]

    // Weighted features using OL weights
    float trend_score = (rsi_val / 100.0) * ol_feature_weights_1 + (1.0 - math.min(1.0, atr_ratio)) * ol_feature_weights_2 + momentum * ol_feature_weights_4

    float range_score = (1.0 - math.abs(rsi_val - 50) / 50) * ol_feature_weights_1 + (1.0 / (1.0 + atr_ratio)) * ol_feature_weights_2

    float vol_score = math.min(1.0, atr_ratio) * ol_feature_weights_2 + (math.abs(momentum) > 0.02 ? 1.0 : 0.0) * ol_feature_weights_4

    // Weighted votes using OL regime weights
    float v_trend = trend_score * ol_regime_weights_trend
    float v_range = range_score * ol_regime_weights_range
    float v_vol = vol_score * ol_regime_weights_vol

    // FIXED: Single-line ternary for regime prediction
    string pred_regime = "RANGING"
    if v_trend > v_range and v_trend > v_vol
        pred_regime := "TRENDING"
    else if v_vol > v_range
        pred_regime := "VOLATILE"

    float max_vote = math.max(v_trend, math.max(v_range, v_vol))
    float total_vote = v_trend + v_range + v_vol
    float confidence = total_vote > 0 ? max_vote / total_vote : 0.0

    [pred_regime, confidence, v_trend, v_range, v_vol]

f_calculate_market_impact(float order_size) =>
    float _total  = 0.0
    float _cost   = 0.0
    float _prate  = participation_rate
    float _klam   = kyle_lambda_current
    float _itemp  = impact_temporary
    float _iperm  = impact_permanent
    float _idecay = impact_decayed
    float _itotal = impact_total
    float _ecost  = expected_impact_cost
    float _ccost  = cumulative_impact_cost
    int   _viols  = impact_violations
    if impact_model_enabled
        float avg_vol  = ta.sma(volume, 20) * 390
        _prate := math.min(avg_vol > 0 ? order_size / avg_vol : 0.0, participation_rate_limit)
        float daily_vol = ta.stdev(close, 20) / ta.sma(close, 20)
        _klam  := kyle_lambda_base * (1.0 + daily_vol * 10.0)
        float lin_imp  = _klam * order_size
        float sqrt_imp = impact_coefficient * daily_vol * math.pow(_prate, impact_alpha)
        float model_imp = impact_model_type == "KyleLinear" ? lin_imp : impact_model_type == "SquareRoot" ? sqrt_imp : (lin_imp + sqrt_imp) / 2.0
        if use_temporary_impact and use_permanent_impact
            _itemp := model_imp * 0.7
            _iperm := model_imp * 0.3
        else if use_temporary_impact
            _itemp := model_imp
            _iperm := 0.0
        else if use_permanent_impact
            _itemp := 0.0
            _iperm := model_imp
        else
            _itemp := model_imp * 0.5
            _iperm := model_imp * 0.5
        _idecay := strategy.position_size != 0 ? _itemp * math.exp(-1.0 * (bar_index - impact_entry_bar) / impact_decay_time) : 0.0
        _itotal := _idecay + _iperm
        _ecost  := _itotal * close
        _ccost  := _ccost + _ecost
        if _prate >= participation_rate_limit
            _viols := _viols + 1
        _total  := _itotal
        _cost   := _ecost
    [_total, _cost, _prate, _klam, _itemp, _iperm, _idecay, _itotal, _ecost, _ccost, _viols]

f_rl_extract_state() =>
    float obi_proxy = (close - low) / (high - low) - 0.5
    float _spread   = (high - low) / close * 100
    float _depth    = volume / (_spread + 0.001)
    float _ofi      = close > open ? volume : -volume
    bool  _toxic    = math.abs(_ofi) > ta.sma(volume, 20) * 2
    float _adverse  = close > close[1] and volume < volume[1] ? 0.5 : close < close[1] and volume > volume[1] ? -0.5 : 0.0
    [obi_proxy, _spread, _depth, _ofi, _toxic, _adverse]

f_rl_policy_forward(float obi, float spread, float depth, float ofi) =>
    // Simple linear policy with 4 weights -> 3 action logits
    float logit_market = rl_policy_weights_1 * obi + rl_policy_weights_2 * spread + rl_policy_weights_3 * depth + rl_policy_weights_4 * ofi
    float logit_limit = -logit_market * 0.5 + rl_policy_weights_1 * spread
    float logit_delay = -math.abs(logit_market) * 0.3

    // Softmax
    float max_logit = math.max(logit_market, math.max(logit_limit, logit_delay))
    float exp_market = math.exp(logit_market - max_logit)
    float exp_limit = math.exp(logit_limit - max_logit)
    float exp_delay = math.exp(logit_delay - max_logit)
    float sum_exp = exp_market + exp_limit + exp_delay

    float prob_market = exp_market / sum_exp
    float prob_limit = exp_limit / sum_exp
    float prob_delay = exp_delay / sum_exp

    [prob_market, prob_limit, prob_delay]

f_rl_value_forward(string regime) =>
    // Regime-aware value estimate
    float base_value = regime == "TRENDING" ? 1.0 : regime == "RANGING" ? 0.5 : 0.0
    base_value + rl_policy_weights_1 * 0.1

f_rl_select_action() =>
    [obi, _sp, _dep, _ofi, _tox, _adv] = f_rl_extract_state()
    [p_market, p_limit, p_delay] = f_rl_policy_forward(obi, _sp, _dep, _ofi)
    float rand_val  = math.random()
    string _action  = "MARKET"
    float  _aprob   = p_market
    if rand_val < p_market
        _action := "MARKET"
        _aprob  := p_market
    else if rand_val < p_market + p_limit
        _action := "LIMIT"
        _aprob  := p_limit
    else
        _action := "DELAY"
        _aprob  := p_delay
    float reward   = strategy.closedtrades > 0 ? strategy.closedtrades.profit(strategy.closedtrades - 1) : 0.0
    float _cumrew  = rl_cumulative_reward * rl_gamma + reward
    float _valest  = f_rl_value_forward(current_regime)
    float _tderr   = reward + rl_gamma * _valest - _valest
    [_action, _aprob, _cumrew, _valest, _tderr, _sp, _dep, _ofi, _tox, _adv]

f_rl_ppo_update() =>
    float _w1  = rl_policy_weights_1
    float _w2  = rl_policy_weights_2
    float _w3  = rl_policy_weights_3
    float _w4  = rl_policy_weights_4
    int   _epc = rl_episode_count
    float _gae = rl_gae_advantage
    float _pl  = rl_policy_loss
    float _ent = rl_entropy
    float _olp = rl_old_log_prob
    float _nlp = rl_new_log_prob
    if rl_enabled and bar_index % rl_update_frequency == 0 and strategy.closedtrades > _epc
        _epc      := strategy.closedtrades
        float adv  = rl_td_error
        _gae      := adv
        float ratio   = _nlp / (_olp + 0.0001)
        float clipped = math.max(1.0 - rl_clip_epsilon, math.min(1.0 + rl_clip_epsilon, ratio))
        _pl  := -math.min(ratio * adv, clipped * adv)
        _ent := -(rl_last_action_prob * math.log(rl_last_action_prob + 0.0001) + (1.0 - rl_last_action_prob) * math.log(1.0 - rl_last_action_prob + 0.0001))
        float upd = rl_learning_rate * _pl
        _w1 := math.max(-1.0, math.min(1.0, _w1 - upd * 0.1 + math.random() * 0.01))
        _w2 := math.max(-1.0, math.min(1.0, _w2 - upd * 0.1 + math.random() * 0.01))
        _w3 := math.max(-1.0, math.min(1.0, _w3 - upd * 0.1 + math.random() * 0.01))
        _w4 := math.max(-1.0, math.min(1.0, _w4 - upd * 0.1 + math.random() * 0.01))
        _olp := _nlp
        _nlp := rl_last_action_prob
    [_w1, _w2, _w3, _w4, _epc, _gae, _pl, _ent, _olp, _nlp]
